#+begin_src emacs-lisp
  (defun ruff-format-buffer ()
    (interactive)
    (let* ((filename (or (buffer-file-name) "buffer.py"))
           (out-buf (generate-new-buffer " *ruff-output*"))
           (exit-code (call-process-region
                       (point-min) (point-max)
                       "ruff" nil out-buf nil
                       "format" "--stdin-filename" filename "-")))
      (if (zerop exit-code)
          (let ((new-content (with-current-buffer out-buf (buffer-string))))
            (if (string= new-content (buffer-substring-no-properties (point-min) (point-max)))
                (message "All checks passed")
              (let ((pos (point)))
                (erase-buffer)
                (insert new-content)
                (goto-char pos))))
        (message "ruff format: failed with exit code %d" exit-code))
      (kill-buffer out-buf)))

  (defun ruff-fix-buffer ()
    (interactive)
    (let* ((original (buffer-substring-no-properties (point-min) (point-max)))
           (tmp-file (make-temp-file "ruff-fix-" nil ".py")))
      (write-region (point-min) (point-max) tmp-file nil 'silent)
      (let ((exit-code (call-process "ruff" nil nil nil
                                     "check" "--fix" tmp-file)))
        (if (zerop exit-code)
            (let ((new-content (with-temp-buffer
                                 (insert-file-contents tmp-file)
                                 (buffer-string))))
              (if (string= new-content original)
                  (message "All checks passed")
                (let ((pos (point)))
                  (erase-buffer)
                  (insert new-content)
                  (goto-char pos))))
          (message "ruff fix: failed with exit code %d" exit-code)))
      (delete-file tmp-file)))

  (defun ruff-on-save ()
    (ruff-fix-buffer)
    (ruff-format-buffer))

  (defun ruff-enable-on-save ()
    (add-hook 'before-save-hook #'ruff-on-save nil t))

  (add-hook 'python-mode-hook #'ruff-enable-on-save)
  (add-hook 'python-ts-mode-hook #'ruff-enable-on-save)
