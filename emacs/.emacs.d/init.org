  (let ((bootstrap-file
	 (expand-file-name
	  "straight/repos/straight.el/bootstrap.el"
	  (or (bound-and-true-p straight-base-dir)
	      user-emacs-directory)))
	(bootstrap-version 7))
    (unless (file-exists-p bootstrap-file)
      (with-current-buffer
	  (url-retrieve-synchronously
	   "https://raw.githubusercontent.com/radian-software/straight.el/develop/install.el"
	   'silent 'inhibit-cookies)
	(goto-char (point-max))
	(eval-print-last-sexp)))
    (load bootstrap-file nil 'nomessage))

  (setq straight-use-package-by-default t)

  ;; Pull recipe repositories only on a clean install — i.e. when at
  ;; least one of the recipe repos hasn't been cloned yet. On an
  ;; existing install this is a no-op, so startup stays fast.
  ;; `org-elpa' is built into straight.el and has no cloned repo, so
  ;; it's excluded from the check.
  (defun my/straight--recipe-repos-ready-p ()
    (let ((repos-dir (expand-file-name
                      "straight/repos/"
                      (or (bound-and-true-p straight-base-dir)
                          user-emacs-directory))))
      (seq-every-p
       (lambda (name)
         (file-directory-p
          (expand-file-name (symbol-name name) repos-dir)))
       (remq 'org-elpa straight-recipe-repositories))))

  (unless (my/straight--recipe-repos-ready-p)
    (straight-pull-recipe-repositories))

#+begin_src emacs-lisp
  (defun ruff-format-buffer ()
    (interactive)
    (let* ((filename (or (buffer-file-name) "buffer.py"))
           (out-buf (generate-new-buffer " *ruff-output*"))
           (exit-code (call-process-region
                       (point-min) (point-max)
                       "ruff" nil out-buf nil
                       "format" "--stdin-filename" filename "-")))
      (if (zerop exit-code)
          (let ((new-content (with-current-buffer out-buf (buffer-string))))
            (if (string= new-content (buffer-substring-no-properties (point-min) (point-max)))
                (message "All checks passed")
              (let ((pos (point)))
                (erase-buffer)
                (insert new-content)
                (goto-char pos))))
        (message "ruff format: failed with exit code %d" exit-code))
      (kill-buffer out-buf)))

  (defun ruff-fix-buffer ()
    (interactive)
    (let* ((original (buffer-substring-no-properties (point-min) (point-max)))
           (tmp-file (make-temp-file "ruff-fix-" nil ".py")))
      (write-region (point-min) (point-max) tmp-file nil 'silent)
      (let ((exit-code (call-process "ruff" nil nil nil
                                     "check" "--fix" tmp-file)))
        (if (zerop exit-code)
            (let ((new-content (with-temp-buffer
                                 (insert-file-contents tmp-file)
                                 (buffer-string))))
              (if (string= new-content original)
                  (message "All checks passed")
                (let ((pos (point)))
                  (erase-buffer)
                  (insert new-content)
                  (goto-char pos))))
          (message "ruff fix: failed with exit code %d" exit-code)))
      (delete-file tmp-file)))

  (defun ruff-on-save ()
    (ruff-fix-buffer)
    (ruff-format-buffer))

  (defun ruff-enable-on-save ()
    (add-hook 'before-save-hook #'ruff-on-save nil t))

** format on save
   Project-aware: walks up from the buffer to find a formatter
   config and invokes the matching tool over stdin. No-ops silently
   when no config is found, so opening a JS/TS file in a project
   that doesn't ship a formatter doesn't surprise you.

#+begin_src emacs-lisp
  (defvar js-ts-biome-configs '("biome.json" "biome.jsonc"))

  (defvar js-ts-prettier-configs
    '(".prettierrc" ".prettierrc.json" ".prettierrc.yaml" ".prettierrc.yml"
      ".prettierrc.json5" ".prettierrc.js" ".prettierrc.cjs"
      ".prettierrc.mjs" ".prettierrc.toml"
      "prettier.config.js" "prettier.config.cjs" "prettier.config.mjs"))

  (defvar js-ts-dprint-configs '("dprint.json" ".dprint.json"))

  (defun js-ts--has-config-p (dir filenames)
    (seq-some (lambda (f) (file-exists-p (expand-file-name f dir))) filenames))

  (defun js-ts--find-config-root (filenames)
    (let ((start (or (and buffer-file-name (file-name-directory buffer-file-name))
                     default-directory)))
      (locate-dominating-file
       start (lambda (d) (js-ts--has-config-p d filenames)))))

  (defun js-ts--resolve-bin (root tool)
    "Prefer ROOT/node_modules/.bin/TOOL when present, else TOOL on $PATH."
    (let ((local (expand-file-name (concat "node_modules/.bin/" tool) root)))
      (if (file-executable-p local) local tool)))

  (defun js-ts--detect-formatter ()
    "Return (CMD ARGS...) for the project's formatter, or nil."
    (let ((file (or (buffer-file-name) "buffer.tsx")))
      (cond
       ((when-let ((root (js-ts--find-config-root js-ts-biome-configs)))
          `(,(js-ts--resolve-bin root "biome")
            "format" "--stdin-file-path" ,file)))
       ((when-let ((root (js-ts--find-config-root js-ts-prettier-configs)))
          `(,(js-ts--resolve-bin root "prettier")
            "--stdin-filepath" ,file)))
       ((when-let ((root (js-ts--find-config-root js-ts-dprint-configs)))
          `(,(js-ts--resolve-bin root "dprint")
            "fmt" "--stdin" ,file))))))

  (defun js-ts-format-buffer ()
    "Format the current buffer with the project's configured formatter."
    (interactive)
    (if-let ((cmd (js-ts--detect-formatter)))
        (let* ((out-buf (generate-new-buffer " *js-ts-format*"))
               (err-file (make-temp-file "js-ts-format-err-"))
               (exit-code (apply #'call-process-region
                                 (point-min) (point-max)
                                 (car cmd) nil (list out-buf err-file) nil
                                 (cdr cmd))))
          (unwind-protect
              (if (zerop exit-code)
                  (let ((new-content (with-current-buffer out-buf (buffer-string)))
                        (old-content (buffer-substring-no-properties
                                      (point-min) (point-max))))
                    (unless (string= new-content old-content)
                      (let ((pos (point)))
                        (erase-buffer)
                        (insert new-content)
                        (goto-char (min pos (point-max))))))
                (message "js-ts format: %s failed (exit %d): %s"
                         (car cmd) exit-code
                         (with-temp-buffer
                           (insert-file-contents err-file)
                           (string-trim (buffer-string)))))
            (kill-buffer out-buf)
            (delete-file err-file)))
      (when (called-interactively-p 'any)
        (message "js-ts: no project formatter config found"))))

  (defun js-ts-enable-format-on-save ()
    (add-hook 'before-save-hook #'js-ts-format-buffer nil t))

  (dolist (h '(js-ts-mode-hook
               typescript-ts-mode-hook
               tsx-ts-mode-hook))
    (add-hook h #'js-ts-enable-format-on-save))

             (todo   . " %i %-14(my/org-agenda-entry-label)")
             (tags   . " %i %-14(my/org-agenda-entry-label)")
             (search . " %i %-14(my/org-agenda-entry-label)")))
   #+end_src

** keybindings
   =C-c c= is plain org-capture; =C-c a= / =C-c v= / =C-c w= / =M-j=
   are the transients / entry points built on top of it.

   #+begin_src emacs-lisp
     (require 'my-org-agenda)
     (require 'my-org-capture)
     (require 'my-org-projects)
     (require 'my-org-refile)
     (global-set-key (kbd "C-c c") #'org-capture)
     (global-set-key (kbd "C-c a") #'my/org-agenda-transient)
     (global-set-key (kbd "C-c v") #'my/org-project-capture)
     (global-set-key (kbd "C-c w") #'my/org-refile-dispatch)
     (global-set-key (kbd "M-j")   #'my/org-project-transient)
   #+end_src

** capture templates
   Rebuild =org-capture-templates= right before each capture (so new
   projects / changed kinds are picked up), and once when org loads so
   the first =C-c c= already has them.

   #+begin_src emacs-lisp
     (require 'my-org-capture)
     (advice-add 'org-capture :before #'my/org-rebuild-capture-templates)

     (with-eval-after-load 'org
       (my/org-rebuild-capture-templates))
   #+end_src

** interview mode
   Auto-enable the interview minor mode in any org buffer that already
   contains an =INTERVIEW_MODE= subtree (on-disk files), and in
   capture buffers once the template has been inserted.

   #+begin_src emacs-lisp
     (require 'my-org-interview)
     (add-hook 'org-mode-hook         #'my/org-interview-maybe-enable)
     (add-hook 'org-capture-mode-hook #'my/org-interview-maybe-enable)
   #+end_src
